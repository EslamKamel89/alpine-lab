// Root state example
<section x-data="{
  todos: [
    { id: 1, title: 'Build the Alpine.js interactions', completed: false },
    { id: 2, title: 'Design the UI', completed: true },
    { id: 3, title: 'Write unit tests', completed: false },
  ],
  newTitle: '',
  filter: 'all',
  get remaining() { return this.todos.filter(t => !t.completed).length },
  get filtered() {
    if (this.filter === 'active') return this.todos.filter(t => !t.completed)
    if (this.filter === 'completed') return this.todos.filter(t => t.completed)
    return this.todos
  }
}">
